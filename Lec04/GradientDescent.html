def error(points, m, b):
    N = len(points)
    TotalError = 0
    for i in range(N):
        x = points[i][0]
        y = points[i][1]
        TotalError += (y - (m*x + b))**2
    return TotalError / float(N)

def move_one_step_down(points, m_current, b_current, learningRate):
    N = float(len(points))
    m_gradient = 0
    b_gradient = 0
    for i in range(len(points)):
        x = points[i][0]
        y = points[i][1]
        m_gradient += (-2/N) * (y - m_current * x - b_current) * (x)
        b_gradient += (-2/N) * (y - m_current * x - b_current)

    new_m = m_current - learningRate * m_gradient
    new_b = b_current - learningRate * b_gradient
    return [new_m, new_b]

def move_down(points, m_initial, b_initial, learningRate, num_iterations):
    N = float(len(points))
    m = m_initial
    b = b_initial
    for i in range(num_iterations):
        m, b = move_one_step_down(points, m, b, learningRate)
        if (i + 1) % 100000 == 0:
            print ("After {0} iterations b = {1}, m = {2}, error = {3}".format(i + 1, b, m, error(points, m, b)))
            plt.plot(points, 'ro')
            plt.axis([-50, 150, -50, 150])
            plt.xlabel('x-axis')
            plt.ylabel('y-axis')
            
            x = np.arange(-50, 150)
            plt.plot(x, m*x + b, '-')
            plt.show()
    
    return m, b

def controls():
    points = np.genfromtxt("data.csv", delimiter=",")
