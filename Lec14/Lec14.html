def AirlineMood(airline, name):
    plt.figure()
    plt.title(name)
    mood_count = airline['airline_sentiment'].value_counts()
    Index = [1, 2, 3]
    plt.bar(Index, mood_count)
    plt.xticks(Index, ['negative', 'neutral', 'positive'])
    plt.ylabel('sentiment count')
    return mood_count
